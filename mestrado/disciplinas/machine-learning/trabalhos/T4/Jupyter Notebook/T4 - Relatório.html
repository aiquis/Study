from sklearn.neural_network import MLPClassifier

def train_neural_network(x, y):
    # Mudando o formato do array para 1d array pois √© o formato que o

N√≥s utilizamos a fun√ß√£o MLPClassifier do pacote scikit-learn para cria√ß√£o da nossa rede completamente conectada. Os par√¢metro para os quais precisamos escolher o melhor valor s√£o os seguintes:

def create_model_neural(input_shape):
    model = Sequential()

    model.add(Dense(2, activation='softmax'))

                        validation_data=(x_test, y_test), shuffle=False)
    return history

from keras.layers import Dense, Conv2D, MaxPooling2D, Dropout, Flatten

def create_model_cnn(input_shape):
    model = Sequential()
    model.add(Conv2D(64, (3, 3), padding='same', activation='relu',
              input_shape=input_shape))
    model.add(MaxPooling2D(pool_size=(3, 3)))
    model.add(Dropout(0.25))

    model.add(Conv2D(128, (3, 3), padding='same', activation='relu'))
    model.add(MaxPooling2D(pool_size=(2, 2)))
    model.add(Dropout(0.25))

    model.add(Flatten())
    model.add(Dense(256, activation='relu'))
    model.add(Dropout(0.5))
    model.add(Dense(2, activation='softmax'))

    return model


def train_model_cnn(model, x_train, y_train, x_test, y_test, batch, epoch):

                  metrics=['accuracy'])

    history = model.fit(x_train, y_train, batch_size=batch, epochs=epoch,
                        validation_data=(x_test, y_test), shuffle=False)
